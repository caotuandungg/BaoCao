# Github config file:  https://github.com/fluent/helm-charts/blob/main/charts/fluent-bit/values.yaml
kind: DaemonSet
hostNetwork: true
dnsPolicy: ClusterFirstWithHostNet

config:
  service: |
    [SERVICE]
        Flush                  5
        Grace                  30
        Log_Level              info
        Daemon                 off
        Parsers_File           parsers.conf
        Parsers_File           custom_parsers.conf
        HTTP_Server            On
        HTTP_Listen            0.0.0.0
        HTTP_Port              2020
        storage.path           /var/fluent-bit/state
        storage.sync           normal
        storage.checksum       off
        storage.max_chunks_up  128

  inputs: |
    [INPUT]
        Name              tail
        Path              /var/log/containers/*_dung-lab_*.log
        Read_from_Head    On
        Parser            cri
        Tag               kube.var.log.containers.*
        Refresh_Interval  5
        Mem_Buf_Limit     10MB
        Skip_Long_Lines   On
        DB                /var/fluent-bit/state/tail-db
        DB.locking        true
        storage.type      filesystem

  filters: |
    [FILTER]
        Name                kubernetes
        Match               kube.*
        Kube_Tag_Prefix     kube.var.log.containers.
        Merge_Log           Off
        K8S-Logging.Parser  On
        K8S-Logging.Exclude Off

    [FILTER]
        Name         parser
        Match        kube.*
        Key_Name     log 
        Parser       json_log
        Reserve_Data On
        Preserve_Key Off

  outputs: |
    [OUTPUT]
        Name           kafka
        Match          kube.*
        Brokers        my-cluster-kafka-bootstrap.kafka-dung.svc:9092
        Topics         dung-logs-topic
        Format         json
        Timestamp_Key  @timestamp
        Timestamp_Format iso8601
        # Giảm disk Fluenbit từ 500MB xuống 50MB 
        storage.total_limit_size 50MB

  customParsers: |
    [PARSER]
        Name        json_log
        Format      json
        Time_Key    @timestamp
        Time_Format %Y-%m-%dT%H:%M:%S.%L%z
        Time_Keep   On

rbac:
  create: true
  nodeAccess: true

daemonSetVolumes:
  - name: varlog
    hostPath:
      path: /var/log
  - name: varlibdockercontainers
    hostPath:
      path: /var/lib/docker/containers
  - name: etcmachineid
    hostPath:
      path: /etc/machine-id
      type: File
  - name: fluent-bit-state
    hostPath:
      path: /var/fluent-bit/state
      type: DirectoryOrCreate

daemonSetVolumeMounts:
  - name: varlog
    mountPath: /var/log
  - name: varlibdockercontainers
    mountPath: /var/lib/docker/containers
    readOnly: true
  - name: etcmachineid
    mountPath: /etc/machine-id
    readOnly: true
  - name: fluent-bit-state
    mountPath: /var/fluent-bit/state
